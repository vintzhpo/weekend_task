import matplotlib.pyplot as plt,import seaborn as sns,"from sklearn.tree import DecisionTreeClassifier, plot_tree"   энийг ашиглан  графикаар харуулж болох  зүйлүүдын сангын зарлаж өгсөн 
x=df[['complaints_count', 'monthly_spend']] энүүгээр загварт орох хувьсагчуудаа өгсөн 
y=df['is_churned'] зорилтод хувьсагчыг олгосон 
clf= DecisionTreeClassifier( max_depth=2,random_state=42)  энүүгээр модны загвар зөвхөн  2  хэсэгт хуваагдан явна гэдэг ерөнхий загвар  зааж өгсөн 
clf.fit(x, y) загвар дээрээ өгөдсөн дата гаа  fit  хийхийн тулд заавал оруулсан 
plt.figure(figsize=(12, 8),dpi=150)  дүрсийн хэмжээгээ 12x8 харьцаатай мөн  150 dpi  ийн нягтаршилтайгаар  харуулхаар тохируулан 
plot_tree(
    clf,
    feature_names=x.columns,       Салбарлах шалгуур болох багануудын нэр
    class_names=['Stay', 'Chur'],  Гаралтын ангилал (Үлдсэн, Цуцалсан)
    filled=True,                   Ангилалаар нь өнгө оруулах
    rounded=True,                  Дөрвөлжин хайрцгийг бөөрөнхий ирмэгтэй болгох
    fontsize=10                    Үсгийн хэмжээ
)
plt.title, plt.show  оороо гарчиг болон  үзүүлэн гээ гаргуулсан
